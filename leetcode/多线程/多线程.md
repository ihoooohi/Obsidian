```go
package main

import (
	"fmt"
)

func a(ca, ab chan bool) {
	for i := 0; i < 5; i++ {
		<- ca
		fmt.Print("a")
		ab <- true
	}
}
func b(ab, bc chan bool) {
	for i := 0; i < 5; i++ {
		<- ab
		fmt.Print("b")
		bc <- true
	}
}
func c(bc,ca, done chan bool) {
	for i := 0; i < 5; i++ {
		<- bc
		fmt.Println("c")
		if i == 4 {
			done <- true
		} else {
			ca <- true
		}
	}
}


func main() {
	ab := make(chan bool)
	bc := make(chan bool)
	ca := make(chan bool)
	done := make(chan bool)
	go a(ca, ab)
	go b(ab, bc)
	go c(bc, ca, done)
	ca <- true
	<- done
}
```
